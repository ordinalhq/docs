---
title: 'Get post'
openapi: 'GET /posts/{id}'
---

Retrieve a specific post by its ID.

## Path parameters

<ParamField path="id" type="string" required>
  UUID of the post to retrieve
</ParamField>

## Response

<ResponseField name="post" type="object">
  The post object
  
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      UUID of the post
    </ResponseField>
    
    <ResponseField name="title" type="string">
      The post title
    </ResponseField>
    
    <ResponseField name="channels" type="array">
      Array of channels this post is configured for
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Current status: `Tentative`, `ToDo`, `InProgress`, `ForReview`, `Blocked`, `Finalized`, `Scheduled`, or `Posted`
    </ResponseField>
    
    <ResponseField name="publishDate" type="string">
      Scheduled publish date (YYYY-MM-DD) in workspace timezone or UTC if not set
    </ResponseField>
    
    <ResponseField name="publishAt" type="datetime">
      Scheduled publish timestamp (UTC). Null if no specific time is set.
    </ResponseField>
    
    <ResponseField name="createdAt" type="datetime">
      Timestamp when the post was created
    </ResponseField>
    
    <ResponseField name="updatedAt" type="datetime">
      Timestamp when the post was last updated
    </ResponseField>
    
    <ResponseField name="linkedIn" type="object">
      LinkedIn content (if configured)
      
      <Expandable title="properties">
        <ResponseField name="copy" type="string">
          The post text content
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
